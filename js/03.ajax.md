# ajax

```
$.ajax({
type: "GET",
url: "여기에URL을입력",
data: {},
success: function(response){
console.log(response)
}
})
```

```
(Get 요청 확인 ajax코드)
$.ajax({
type: "GET",
url: "/test?title_give=봄날은간다",
data: {},

console.log(response)
}
})

(Post 요청 확인 ajax코드)
$.ajax({
type: "POST",
url: "/test",
data: { title_give:'봄날은간다' },
success: function(response){
console.log(response)
}
})
```
